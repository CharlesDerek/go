   264  func fromEnv(env []string) map[string]string {
   265  	m := make(map[string]string)
   266  	for _, envPair := range env {
   267  		equalIndex := strings.IndexRune(envPair, '=')
   268  		if equalIndex == -1 {
   269  			panic("Invalid env key-value pair: " + envPair)
   270  		}
   271  		key, value := envPair[:equalIndex], envPair[equalIndex+1:]
   272  		m[key] = value
   273  	}
   274  	return m
   275  }
