    45  func readMacOSDNS(ctx context.Context, getSCUtilDNS scutilExecutor) (Config, error) {
    46  	output, err := getSCUtilDNS(ctx)
    47  	if err != nil {
    48  		return Config{}, err
    49  	}

    51  	scanner := bufio.NewScanner(bytes.NewReader(output))
    52  	var config Config
    53  	for scanner.Scan() {
    54  		key, value := splitKeyValue(scanner.Text())
    55  		if strings.Contains(key, "resolver") {
    56  			config.Resolvers = append(config.Resolvers, Resolver{})
    57  			continue
    58  		}
    59  		if len(config.Resolvers) == 0 {
    60  			continue
    61  		}
    62  
    63  		currentResolver := &config.Resolvers[len(config.Resolvers)-1]
    64  		switch {
    65  		case strings.Contains(key, "search domain"):
    66  			currentResolver.SearchDomain = append(currentResolver.SearchDomain, value)
    67  		case strings.Contains(key, "domain"):
    68  			currentResolver.Domain = value
    69  		case strings.Contains(key, "flags"):
    70  			for _, flag := range strings.Split(value, ",") {
    71  				currentResolver.Flags = append(currentResolver.Flags, Flag(strings.TrimSpace(flag)))
    72  			}
    73  		case strings.Contains(key, "if_index"):
    74  			tokens := strings.SplitN(value, " ", 2)
    75  			if len(tokens) == 2 {
    76  				currentResolver.InterfaceName = strings.Trim(tokens[1], "()")
    77  			}
    78  
    79  			i, err := strconv.ParseInt(tokens[0], 10, 64)
    80  			if err == nil {
    81  				currentResolver.InterfaceIndex = int(i)
    82  			}
    83  		case strings.Contains(key, "options"):
    84  			currentResolver.MulticastDNS = strings.Contains(value, "mdns")
    85  		case strings.Contains(key, "port"):
    86  			i, err := strconv.ParseInt(value, 10, 64)
    87  			if err == nil {
    88  				currentResolver.Port = int(i)
    89  			}
    90  		case strings.Contains(key, "nameserver"):
    91  			currentResolver.Nameservers = append(currentResolver.Nameservers, value)
    92  		case strings.Contains(key, "order"):
    93  			i, err := strconv.ParseInt(value, 10, 64)
    94  			if err == nil {
    95  				currentResolver.Order = int(i)
    96  			}
    97  		case strings.Contains(key, "reach"):
    98  			tokens := strings.SplitN(value, " ", 2)
    99  			if len(tokens) == 2 {
   100  				reach := strings.Trim(tokens[1], "()")
   101  				for _, statusStr := range strings.Split(reach, ",") {
   102  					status := Reach(statusStr)
   103  					if status == Reachable {
   104  						currentResolver.reachable = true
   105  					}
   106  					currentResolver.Reach = append(currentResolver.Reach, status)
   107  				}
   108  			}
   109  		case strings.Contains(key, "timeout"):
   110  			i, err := strconv.ParseInt(value, 10, 64)
   111  			if err == nil {
   112  				currentResolver.Timeout = time.Duration(i) * time.Second
   113  			}
   114  		}
   115  	}
   116  
   117  	return config, nil
   118  }
