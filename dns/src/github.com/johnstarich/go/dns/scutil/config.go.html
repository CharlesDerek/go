    44  func readMacOSDNS(ctx context.Context, getSCUtilDNS scutilExecutor) (Config, error) {
    45  	output, err := getSCUtilDNS(ctx)
    46  	if err != nil {
    47  		return Config{}, err
    48  	}
    49  
    50  	scanner := bufio.NewScanner(bytes.NewReader(output))
    51  	var config Config
    52  	for scanner.Scan() {
    53  		key, value := splitKeyValue(scanner.Text())
    54  		if strings.Contains(key, "resolver") {
    55  			config.Resolvers = append(config.Resolvers, Resolver{})
    56  			continue
    57  		}
    58  		if len(config.Resolvers) == 0 {
    59  			continue
    60  		}
    61  
    62  		currentResolver := &config.Resolvers[len(config.Resolvers)-1]
    63  		switch {
    64  		case strings.Contains(key, "search domain"):
    65  			currentResolver.SearchDomain = append(currentResolver.SearchDomain, value)
    66  		case strings.Contains(key, "domain"):
    67  			currentResolver.Domain = value
    68  		case strings.Contains(key, "flags"):
    69  			for _, flag := range strings.Split(value, ",") {
    70  				currentResolver.Flags = append(currentResolver.Flags, Flag(strings.TrimSpace(flag)))
    71  			}
    72  		case strings.Contains(key, "if_index"):
    73  			tokens := strings.SplitN(value, " ", 2)
    74  			if len(tokens) == 2 {
    75  				currentResolver.InterfaceName = strings.Trim(tokens[1], "()")
    76  			}
    77  
    78  			i, err := strconv.ParseInt(tokens[0], 10, 64)
    79  			if err == nil {
    80  				currentResolver.InterfaceIndex = int(i)
    81  			}
    82  		case strings.Contains(key, "options"):
    83  			currentResolver.MulticastDNS = strings.Contains(value, "mdns")
    84  		case strings.Contains(key, "nameserver"):
    85  			currentResolver.Nameservers = append(currentResolver.Nameservers, value)
    86  		case strings.Contains(key, "order"):
    87  			i, err := strconv.ParseInt(value, 10, 64)
    88  			if err == nil {
    89  				currentResolver.Order = int(i)
    90  			}
    91  		case strings.Contains(key, "reach"):
    92  			tokens := strings.SplitN(value, " ", 2)
    93  			if len(tokens) == 2 {
    94  				reach := strings.Trim(tokens[1], "()")
    95  				for _, statusStr := range strings.Split(reach, ",") {
    96  					status := Reach(statusStr)
    97  					if status == Reachable {
    98  						currentResolver.reachable = true
    99  					}
   100  					currentResolver.Reach = append(currentResolver.Reach, status)
   101  				}
   102  			}
   103  		case strings.Contains(key, "timeout"):
   104  			i, err := strconv.ParseInt(value, 10, 64)
   105  			if err == nil {
   106  				currentResolver.Timeout = time.Duration(i) * time.Second
   107  			}
   108  		}
   109  	}
   110  
   111  	return config, nil
   112  }

   114  func splitKeyValue(line string) (key, value string) {

   116  	switch len(tokens) {
