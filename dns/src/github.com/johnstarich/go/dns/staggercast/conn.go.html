    56  	if len(conns) == 0 {
    57  		panic("connection count must be non-zero")

    76  // Stagger controls when secondary connections are attempted after the first. For any new enabled connection, the last Write and SetDeadlines are replayed onto it.
    77  // If Stagger is called after the connection is started, then the behavior is undefined.
    78  // First connection is attempted immediately, then all future connections will be enabled when receiving from ticker.
    79  //
    80  // Example: Create and wrap a 'time.Ticker.C' with a duration of 1 second.
    81  // The first connection is attempted immediately. For every 1 second the Read hasn't returned, an additional connection is attempted.
    82  // Once a connection succeeds, the result is returned immediately from Read.
    83  func (s *staggerConn) Stagger(ticker <-chan struct{}, cancel context.CancelFunc) {
    84  	s.replayMu.Lock()
    85  
    86  	ctx, tickerCancel := context.WithCancel(context.Background())
    87  	s.tickerCancel = tickerCancel
    88  
    89  	totalLength := len(s.conns)
    90  	s.replay = make([]chan struct{}, totalLength)
    91  	for i := range s.replay {
    92  		s.replay[i] = make(chan struct{})
    93  	}
    94  	close(s.replay[0]) // initial connection should immediately fire
    95  	connCount := atomic.NewUint64(1)
    96  	s.connCount = connCount
    97  
    98  	// listen for ticks and fire off replays on new connections
    99  	go func() {
   100  		// only reference params and local vars here, no state from 'p'
   101  		for {
   102  			select {
   103  			case <-ticker:
   104  				count := connCount.Inc()
   105  				if count > uint64(totalLength) {
   106  					// finished enabling all connections
   107  					cancel()
   108  					return
   109  				}
   110  				go s.runReplay(count - 1)
   111  			case <-ctx.Done():
   112  				cancel()
   113  				return
   114  			}
   115  		}
   116  	}()
   117  	s.replayMu.Unlock()
   118  }

   143  // runReplay synchronously reapplies the last Deadlines and the last Write on a best-effort basis
   144  func (s *staggerConn) runReplay(connIndex uint64) {
   145  	s.replayMu.RLock()
   146  	if !s.lastDeadline.IsZero() {
   147  		_ = s.conns[connIndex].SetDeadline(s.lastDeadline)
   148  	}
   149  	if !s.lastReadDeadline.IsZero() {
   150  		_ = s.conns[connIndex].SetReadDeadline(s.lastReadDeadline)
   151  	}
   152  	if !s.lastWriteDeadline.IsZero() {
   153  		_ = s.conns[connIndex].SetWriteDeadline(s.lastWriteDeadline)
   154  	}
   155  	b := s.lastWrite
   156  	s.replayMu.RUnlock()
   157  	if b != nil {
   158  		_, _ = s.conns[connIndex].Write(b)
   159  	}
   160  	close(s.replay[connIndex]) // fire off any pending iter's
   161  }

   171  func (s *staggerConn) iter(op connOp, fn func(conn PacketConn) (keepGoing bool, err error)) error {
   172  	ctx, cancel := context.WithCancel(context.Background())
   173  	defer cancel()
   174  	done := make(chan struct{}, len(s.conns))
   175  	errs := make(chan error, len(s.conns))
   176  	for ix, conn := range s.conns {
   177  		go func(ix int, conn PacketConn) {
   178  			select {
   179  			case <-s.replay[ix]:
   180  			case <-ctx.Done():
   181  				return
   182  			}
   183  
   184  			keepGoing, err := fn(conn)
   185  			if err != nil {
   186  				errs <- err
   187  			} else if op == readOp || op == readFromOp {
   188  				s.firstResponder.CAS(-1, int64(ix))
   189  			}
   190  			done <- struct{}{}
   191  			if !keepGoing {
   192  				cancel()
   193  			}
   194  		}(ix, conn)
   195  	}
   196  	for i := 0; i < s.getConnCount(); i++ {
   197  		select {
   198  		case <-done:
   199  		case <-ctx.Done():
   200  			goto ctxDone
   201  		}
   202  	}
   203  
   204  ctxDone:
   205  	if len(errs) >= s.getConnCount() {
   206  		// only return an error if all conns failed
   207  		err := <-errs
   208  		return errors.Wrapf(err, "all connections have failed for %q", op)
   209  	}
   210  	return nil
   211  }

   213  func (s *staggerConn) Read(b []byte) (n int, err error) {
   214  	type byteResp struct {
   215  		n int
   216  		b []byte
   217  	}
   218  	success := make(chan byteResp, len(s.conns))
   219  	failure := make(chan byteResp, len(s.conns))
   220  	err = s.iter(readOp, func(conn PacketConn) (bool, error) {
   221  		buf := make([]byte, len(b))
   222  		n, err := conn.Read(buf)
   223  		if err == nil {
   224  			success <- byteResp{n: n, b: buf}
   225  		} else {
   226  			failure <- byteResp{n: n, b: buf}
   227  		}
   228  		return err != nil, err
   229  	})
   230  	select {
   231  	case resp := <-success:
   232  		n = resp.n
   233  		copy(b, resp.b)
   234  	case resp := <-failure:
   235  		n = resp.n
   236  		copy(b, resp.b)
   237  	}
   238  	return
   239  }

   252  			failure <- n
   253  		}

   257  	case n = <-success:
   258  	case n = <-failure:
   259  	}

   319  func (s *staggerConn) Close() error {
   320  	s.replayMu.RLock()
   321  	s.tickerCancel()
   322  	s.replayMu.RUnlock()
   323  	var foundErr error
   324  	for _, conn := range s.conns {
   325  		err := conn.Close()
   326  		if err != nil {
   327  			foundErr = err
   328  		}
   329  	}
   330  	return foundErr
