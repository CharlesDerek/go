    76  func (m *macOSDialer) ensureNameservers() ([]string, error) {
    77  	m.nameserversMu.RLock()
    78  	nameservers := m.nameservers
    79  	m.nameserversMu.RUnlock()
    80  	if len(nameservers) != 0 {
    81  		return nameservers, nil
    82  	}
    83  	m.nameserversMu.Lock()
    84  	defer m.nameserversMu.Unlock()
    85  	if len(m.nameservers) != 0 { // check again, could change while waiting on lock
    86  		return m.nameservers, nil
    87  	}
    88  
    89  	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
    90  	defer cancel()
    91  	m.Logger.Info("Reading macOS DNS config from 'scutil'...")
    92  	cfg, err := m.readResolvers(ctx)
    93  	for _, resolver := range cfg.Resolvers {
    94  		for _, nameserver := range resolver.Nameservers {
    95  			m.nameservers = append(m.nameservers, nameserver+":53")
    96  		}
    97  	}
    98  	m.Logger.Info("Finished reading macOS DNS config from 'scutil'", zap.Error(err))
    99  	return m.nameservers, err
   100  }

   102  func (m *macOSDialer) DialContext(ctx context.Context, network, address string) (net.Conn, error) {
   103  	nameservers, err := m.ensureNameservers()
   104  	if err != nil {
   105  		m.Logger.Error("Failed looking up macOS resolvers, falling back to builtin DNS dialer", zap.Error(err))
   106  		return m.dialer.DialContext(ctx, network, address)
   107  	}
   108  
   109  	conn, err := m.dialAll(ctx, nameservers)
   110  	if err != nil {
   111  		m.Logger.Error("Failed dialing macOS nameservers, falling back to builtin DNS dialer", zap.Error(err))
   112  		return m.dialer.DialContext(ctx, network, address)
   113  	}
   114  
   115  	return conn, nil
   116  }

   118  func (m *macOSDialer) dialAll(ctx context.Context, nameservers []string) (net.Conn, error) {
   119  	type dialResp struct {
   120  		ix   int
   121  		conn net.Conn
   122  	}
   123  	conns := make(chan dialResp, len(nameservers))
   124  
   125  	var wait sync.WaitGroup
   126  	wait.Add(len(nameservers))
   127  
   128  	m.Logger.Debug("Dialing all nameservers")
   129  	for ix, nameserver := range nameservers {
   130  		go func(ix int, nameserver string) {
   131  			conn, err := m.dialer.DialContext(ctx, "udp", nameserver)
   132  			if err != nil {
   133  				m.Logger.Warn("Error dialing nameserver", zap.String("nameserver", nameserver), zap.Int("index", ix), zap.Error(err))
   134  			} else {
   135  				conns <- dialResp{ix: ix, conn: conn}
   136  			}
   137  			wait.Done()
   138  		}(ix, nameserver)
   139  	}
   140  	wait.Wait()
   141  	select {
   142  	case <-ctx.Done():
   143  		return nil, ctx.Err()
   144  	default:
   145  	}
   146  	close(conns)
   147  
   148  	var allSuccessfulResps []dialResp
   149  	for resp := range conns {
   150  		allSuccessfulResps = append(allSuccessfulResps, resp)
   151  	}
   152  	if len(allSuccessfulResps) == 0 {
   153  		return nil, errors.New("error dialing all nameservers")
   154  	}
   155  	sort.Slice(allSuccessfulResps, func(a, b int) bool { // preserve original nameserver order
   156  		return allSuccessfulResps[a].ix < allSuccessfulResps[b].ix
   157  	})
   158  
   159  	var allSuccessfulConns []staggercast.PacketConn
   160  	for _, resp := range allSuccessfulResps {
   161  		allSuccessfulConns = append(allSuccessfulConns, resp.conn.(staggercast.PacketConn)) // UDP connections must also support net.PacketConn
   162  	}
   163  
   164  	staggerConn := staggercast.New(allSuccessfulConns)
   165  	staggerConn.Stagger(staggerTicker(m.InitialNameserverDelay, m.NextNameserverInterval, m.Logger))
   166  	go m.reorderNameservers(ctx, staggerConn)
   167  	return staggerConn, nil
   168  }

   170  func staggerTicker(initialDelay, d time.Duration, logger *zap.Logger) (<-chan struct{}, context.CancelFunc) {
   171  	ctx, cancel := context.WithCancel(context.Background())
   172  	c := make(chan struct{})
   173  	go func() {
   174  		index := 1
   175  		timer := time.NewTimer(initialDelay)
   176  		select {
   177  		case <-timer.C:
   178  			logger.Debug("scatter: Enabling connection", zap.Int("index", index))
   179  			index++
   180  			c <- struct{}{}
   181  		case <-ctx.Done():
   182  			timer.Stop()
   183  			return
   184  		}
   185  		timer.Stop()
   186  
   187  		ticker := time.NewTicker(d)
   188  		for {
   189  			select {
   190  			case <-ctx.Done():
   191  				ticker.Stop()
   192  				close(c)
   193  				return
   194  			case <-ticker.C:
   195  				logger.Debug("scatter: Enabling connection", zap.Int("index", index))
   196  				index++
   197  				c <- struct{}{}
   198  			}
   199  		}
   200  	}()
   201  	return c, cancel
   202  }

   204  func (m *macOSDialer) reorderNameservers(ctx context.Context, conn staggercast.Conn) {
   205  	var zero time.Time
   206  	if deadline, ok := ctx.Deadline(); !ok || deadline == zero {
   207  		m.Logger.Debug("Skipping nameserver reorder, no deadline on context")
   208  		return
   209  	}
   210  
   211  	<-ctx.Done()
   212  	stats := conn.Stats()
   213  	if stats.FastestRemoteIndex == 0 {
   214  		return
   215  	}
   216  
   217  	m.nameserversMu.Lock()
   218  	defer m.nameserversMu.Unlock()
   219  
   220  	fastestRemoteAddr := stats.FastestRemote.String()
   221  	fastIndexNS := m.nameservers[stats.FastestRemoteIndex]
   222  	if fastIndexNS != fastestRemoteAddr {
   223  		m.Logger.Debug("Fastest remote already reordered", zap.String("remote", stats.FastestRemote.String()), zap.String("nsIndex", fastIndexNS))
   224  		return
   225  	}
   226  	newNS := []string{fastIndexNS}
   227  	newNS = append(newNS, m.nameservers[:stats.FastestRemoteIndex]...)
   228  	newNS = append(newNS, m.nameservers[stats.FastestRemoteIndex+1:]...)
   229  	m.nameservers = newNS
   230  	m.Logger.Debug("Reordering fastest nameserver to the front", zap.Strings("nameservers", newNS))
   231  }
