   117  func formatGraph(f float64, format summaryFormat) string {
   118  	const (
   119  		graphWidth    = 5
   120  		graphTickSize = 1.0 / graphWidth
   121  	)
   122  	total := graphWidth
   123  	var graph strings.Builder
   124  	if f == 0 {
   125  		graph.WriteRune(percentRune(0))
   126  		total--
   127  	}
   128  	for f > 0 {
   129  		graph.WriteRune(percentRune(f / graphTickSize))
   130  		f -= graphTickSize
   131  		total--
   132  	}
   133  	for total > 0 {
   134  		graph.WriteRune(' ')
   135  		total--
   136  	}
   137  	return format.Colorize(boldColor, graph.String())
