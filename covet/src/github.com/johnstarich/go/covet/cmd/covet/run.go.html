   123  func runArgs(args Args, deps Deps) (err error) {
   124  	defer func() { err = errors.WithStack(err) }()
   125  
   126  	var diffFile io.Reader
   127  	if args.DiffFile == "-" {
   128  		diffFile = deps.Stdin
   129  	} else {
   130  		f, err := deps.FS.Open(args.DiffFile)
   131  		if err != nil {
   132  			return err
   133  		}
   134  		defer f.Close()
   135  		diffFile = f
   136  	}
   137  
   138  	covet, err := covet.Parse(covet.Options{
   139  		FS:             deps.FS,
   140  		Diff:           diffFile,
   141  		DiffBaseDir:    args.DiffBaseDir,
   142  		GoCoveragePath: args.GoCoverageFile,
   143  	})
   144  	if err != nil {
   145  		return err
   146  	}
   147  	if len(covet.DiffCoverageFiles()) == 0 {
   148  		fmt.Fprintln(deps.Stdout, "No coverage information intersects with diff.")
   149  		return nil
   150  	}
   151  
   152  	totalCovered := covet.DiffCovered()
   153  
   154  	uncoveredFiles := findReportableUncoveredFiles(covet.DiffCoverageFiles(), float64(args.TargetDiffCoverage)/maxPercentInt, totalCovered)
   155  
   156  	if args.ShowCoverage {
   157  		for _, f := range uncoveredFiles {
   158  			fmt.Fprintln(deps.Stdout, "Coverage diff:", f.Name)
   159  			err := printCovet(deps.Stdout, deps.FS, f, args.GoCoverageFile)
   160  			if err != nil {
   161  				return err
   162  			}
   163  		}
   164  	}
   165  
   166  	fmt.Fprintln(deps.Stdout)
   167  	totalCoveredStatus := newCoverageStatus(totalCovered)
   168  	fmt.Fprintln(deps.Stdout, "Total diff coverage:", totalCoveredStatus.Colorize(formatPercent(totalCovered)))
   169  	fmt.Fprintln(deps.Stdout)
   170  	summary := covetSummary(uncoveredFiles, args.TargetDiffCoverage, summaryTable)
   171  	fmt.Fprint(deps.Stdout, summary)
   172  
   173  	runWorkflow(coverageCommand(totalCovered, "", nil))
   174  	for _, f := range uncoveredFiles {
   175  		runWorkflow(coverageCommand(coveredFile(f), f.Name, findUncoveredLines(f)))
   176  	}
   177  
   178  	if args.GitHubToken != "" {
   179  		org, repo, number, err := parseIssueURL(args.GitHubIssue)
   180  		if err != nil {
   181  			return err
   182  		}
   183  		err = ensureAppGitHubComment(context.Background(), gitHubCommentOptions{
   184  			GitHubEndpoint: args.GitHubEndpoint,
   185  			GitHubToken:    args.GitHubToken,
   186  			RepoOwner:      org,
   187  			Repo:           repo,
   188  			IssueNumber:    number,
   189  			Body:           covetSummary(uncoveredFiles, args.TargetDiffCoverage, summaryMarkdown),
   190  		})
   191  		if err != nil {
   192  			fmt.Fprintln(deps.Stdout, "\nFailed to update GitHub comment, skipping. Error:", err)
   193  		}
   194  	}
   195  	return nil
   196  }

   198  func printCovet(w io.Writer, fs hackpadfs.FS, f covet.File, covPath string) error {
   199  	r, err := openFile(fs, f.Name, covPath)
   200  	if err != nil {
   201  		return err
   202  	}
   203  	defer r.Close()
   204  
   205  	chunks, err := covet.DiffChunks(f, r)
   206  	if err != nil {
   207  		return err
   208  	}
   209  	for _, chunk := range chunks {
   210  		fmt.Fprintln(w, "Coverage:", chunk.FirstLine, "to", chunk.LastLine)
   211  		for _, line := range chunk.Lines {
   212  			switch {
   213  			case strings.HasPrefix(line, "+"):
   214  				line = color.GreenString(line)
   215  			case strings.HasPrefix(line, "-"):
   216  				line = color.RedString(line)
   217  			}
   218  			fmt.Fprintln(w, line)
   219  		}
   220  	}
   221  	return nil
   222  }

   233  func findUncoveredLines(f covet.File) []span.Span {
   234  	var uncoveredLines []span.Span
   235  	ok := true
   236  	var nextLineIndex int
   237  	for ok {
   238  		var uncovered span.Span
   239  		uncovered, ok, nextLineIndex = findFirstUncoveredLines(f.Lines, nextLineIndex)
   240  		if ok {
   241  			uncoveredLines = append(uncoveredLines, uncovered)
   242  		}
   243  	}
   244  	sort.SliceStable(uncoveredLines, func(a, b int) bool {
   245  		return uncoveredLines[a].Len() > uncoveredLines[b].Len()
   246  	})
   247  	return uncoveredLines
   248  }

   250  func findFirstUncoveredLines(lines []covet.Line, startIndex int) (uncovered span.Span, ok bool, nextLineIndex int) {
   251  	// find start
   252  	nextLineIndex = startIndex
   253  	for _, l := range lines[nextLineIndex:] {
   254  		nextLineIndex++
   255  		if !l.Covered {
   256  			n := int64(l.LineNumber)
   257  			uncovered = span.Span{
   258  				Start: n,
   259  				End:   n + 1,
   260  			}
   261  			ok = true
   262  			break
   263  		}
   264  	}
   265  	// find next line number jump or covered line
   266  	for _, l := range lines[nextLineIndex:] {
   267  		if l.Covered || int64(l.LineNumber) != uncovered.End {
   268  			break
   269  		}
   270  		nextLineIndex++
   271  		uncovered.End++
   272  	}
   273  	return
   274  }

   276  func findReportableUncoveredFiles(coveredFiles []covet.File, target, current float64) []covet.File {
   277  	// sort by highest uncovered line count
   278  	sort.Slice(coveredFiles, func(aIndex, bIndex int) bool {
   279  		a, b := coveredFiles[aIndex], coveredFiles[bIndex]
   280  		switch {
   281  		case a.Uncovered != b.Uncovered:
   282  			return a.Uncovered > b.Uncovered
   283  		default:
   284  			return a.Name < b.Name
   285  		}
   286  	})
   287  
   288  	var uncoveredFiles []covet.File
   289  	// find minimum number of covered lines required to hit target
   290  	targetMissingLines := 0
   291  	totalLines := uint(0)
   292  	for _, f := range coveredFiles {
   293  		totalLines += uint(f.Covered + f.Uncovered)
   294  	}
   295  	if percentDiff := target - current; percentDiff > 0 {
   296  		targetMissingLines = int(percentDiff * float64(totalLines))
   297  	} else {
   298  		return nil // target is met
   299  	}
   300  	// next, collect the biggest uncovered files until we'd hit the target
   301  	for _, f := range coveredFiles {
   302  		const minUncoveredThreshold = 2 // include more files if it is slim pickings
   303  		if f.Uncovered > 0 {
   304  			uncoveredFiles = append(uncoveredFiles, f)
   305  		}
   306  		if f.Uncovered > minUncoveredThreshold {
   307  			targetMissingLines -= int(f.Uncovered)
   308  		}
   309  		if targetMissingLines <= 0 {
   310  			break
   311  		}
   312  	}
   313  	return uncoveredFiles
   314  }

   321  func parseIssueURL(s string) (org, repo string, number int, err error) {
   322  	if s == "" {
   323  		err = fmt.Errorf("-gh-issue is required")
   324  		return
   325  	}
   326  	issueURL, err := url.Parse(s)
   327  	if err != nil {
   328  		return
   329  	}
   330  	if issueURL.Scheme == "" {
   331  		issueURL, err = url.Parse("https://" + s)
   332  		if err != nil {
   333  			return
   334  		}
   335  	}
   336  	const minIssueURLPathComponents = 4
   337  	tokens := strings.SplitN(strings.TrimPrefix(issueURL.Path, "/"), "/", minIssueURLPathComponents+1)
   338  	if len(tokens) < minIssueURLPathComponents {
   339  		err = fmt.Errorf("malformed issue URL: expected 4+ path components, e.g. github.com/org/repo/pull/123")
   340  		return
   341  	}
   342  	n, err := strconv.ParseInt(tokens[3], decimalBase, maxIntBits)
   343  	if err != nil {
   344  		return
   345  	}
   346  	org = tokens[0]
   347  	repo = tokens[1]
   348  	number = int(n)
   349  	return
   350  }
