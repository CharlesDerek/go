    18  func (f File) findContextSpans(contextLines uint) []span.Span {
    19  	if len(f.Lines) == 0 {
    20  		return nil
    21  	}
    22  
    23  	var overlappingSpans []span.Span
    24  	for _, line := range f.Lines {
    25  		overlappingSpans = append(overlappingSpans, span.Span{
    26  			Start: max(1, int64(line.LineNumber)-int64(contextLines)),
    27  			End:   int64(line.LineNumber + contextLines + 1), // may be beyond end of file, but can stop line iteration at EOF
    28  		})
    29  	}
    30  	spans := []span.Span{overlappingSpans[0]}
    31  	for _, s := range overlappingSpans {
    32  		lastSpan := &spans[len(spans)-1]
    33  		newSpan, merged := lastSpan.Merge(s)
    34  		if merged {
    35  			*lastSpan = newSpan
    36  		} else {
    37  			spans = append(spans, s)
    38  		}
    39  	}
    40  	return spans
    41  }

    65  // DiffChunks return diff-like Chunks from a covet.File and the file contents' Reader.
    66  func DiffChunks(file File, fileReader io.Reader) ([]DiffChunk, error) {
    67  	var chunks []DiffChunk
    68  	iter := newLineIterator(fileReader)
    69  	const contextLines = 2
    70  	var lineNumber int64 = 1
    71  	diffLineIndex := 0
    72  	for _, s := range file.findContextSpans(contextLines) {
    73  		if lineNumber < s.Start {
    74  			err := iter.SkipLines(s.Start - lineNumber)
    75  			if err != nil {
    76  				return nil, err
    77  			}
    78  			lineNumber = s.Start
    79  		}
    80  		lines, err := iter.NextLines(s.Len())
    81  		if err != nil {
    82  			return nil, err
    83  		}
    84  		for i := range lines {
    85  			op := noOpPrefix
    86  			if diffLineIndex < len(file.Lines) {
    87  				diffLine := file.Lines[diffLineIndex]
    88  				if lineNumber == int64(diffLine.LineNumber) {
    89  					op = diffLine.diffOpPrefix()
    90  					diffLineIndex++
    91  				}
    92  			}
    93  			lines[i] = op + lines[i]
    94  			lineNumber++
    95  		}
    96  		chunks = append(chunks, DiffChunk{
    97  			FirstLine: uint(s.Start),
    98  			LastLine:  uint(lineNumber - 1),
    99  			Lines:     lines,
   100  		})
   101  	}
   102  	return chunks, nil
   103  }
