    13  // New generates a new summary report in the given format
    14  func New(uncoveredFiles []coverfile.File, targetCoverage uint, format Format) string {
    15  	if len(uncoveredFiles) == 0 {
    16  		return fmt.Sprintf("Successfully reached diff coverage target: >%d%%\n", targetCoverage)
    17  	}
    18  
    19  	var sb strings.Builder
    20  	sb.WriteString("Diff coverage is below target. Add tests for these files:\n")
    21  	tbl := table.NewWriter()
    22  	const coverageColumnIndex = 2
    23  	tbl.SetColumnConfigs([]table.ColumnConfig{
    24  		{Number: coverageColumnIndex, Align: text.AlignCenter},
    25  	})
    26  	tbl.SuppressEmptyColumns()
    27  	bold := boldColor()
    28  	tbl.AppendHeader(table.Row{
    29  		"",
    30  		format.Colorize(bold, "Lines"),
    31  		format.Colorize(bold, "Coverage"),
    32  		format.Colorize(bold, "File"),
    33  	})
    34  	for _, f := range uncoveredFiles {
    35  		percent := FileCoverage(f)
    36  		status := coverstatus.New(percent)
    37  		tbl.AppendRow(table.Row{
    38  			format.StatusIcon(status),
    39  			format.ColorizeStatus(status, format.Monospace(formatFraction(f.Covered, f.Uncovered+f.Covered))),
    40  			format.ColorizeStatus(status, format.Monospace(FormatPercent(percent)+" "+formatGraph(percent, format))),
    41  			f.Name,
    42  		})
    43  	}
    44  	sb.WriteString(format.FormatTable(tbl))
    45  	sb.WriteRune('\n')
    46  	return sb.String()
    47  }
