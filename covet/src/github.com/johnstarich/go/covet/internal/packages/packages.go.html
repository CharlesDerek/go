    98  	ctx := build.Default

   100  	ctx.GOPATH = options.GoPath

   102  	ctx.SplitPathList = filepath.SplitList
   103  	ctx.IsAbsPath = func(path string) bool {
   104  		return !build.IsLocalImport(path)
   105  	}
   106  	ctx.IsDir = func(path string) bool {
   107  		info, err := hackpadfs.Stat(fs, path)
   108  		return err == nil && info.IsDir()
   109  	}
   110  	ctx.HasSubdir = func(root, dir string) (rel string, ok bool) {
   111  		// TODO add EvalSymlinks support to hackpadfs
   112  		const sep = "/"
   113  		root = path.Clean(root)
   114  		if !strings.HasSuffix(root, sep) {

   116  		}
   117  		dir = path.Clean(dir)
   118  		if !strings.HasPrefix(dir, root) {
   119  			return "", false
   120  		}
   121  		return dir[len(root):], true
   122  	}
   123  	ctx.ReadDir = fsReadDir(fs)
   124  	ctx.OpenFile = func(path string) (io.ReadCloser, error) {
   125  		return fs.Open(path)
   126  	}
   127  	return ctx

   133  	return func(dir string) ([]hackpadfs.FileInfo, error) {
   134  		dirEntries, err := hackpadfs.ReadDir(fs, dir)
   135  		if err != nil {

   138  		var infos []hackpadfs.FileInfo

   162  func getModule(fs hackpadfs.FS, dir string) (moduleName, moduleDir string, ok bool, err error) {
   163  	file := path.Join(dir, "go.mod")
   164  	_, err = hackpadfs.Stat(fs, file)
   165  	if err != nil {
   166  		if errors.Is(err, hackpadfs.ErrNotExist) {
   167  			err = nil
   168  		}
   169  		return "", "", false, err
   170  	}
   171  
   172  	contents, err := hackpadfs.ReadFile(fs, file)
   173  	if err != nil {
   174  		return "", "", false, err
   175  	}
   176  	modFile, err := modfile.Parse(file, contents, nil)
   177  	if err != nil {
   178  		return "", "", false, err
   179  	}
   180  	return modFile.Module.Mod.Path, dir, true, nil
