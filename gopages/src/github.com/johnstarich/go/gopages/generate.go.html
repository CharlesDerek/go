    56  func addGoPagesFuncs(funcs template.FuncMap, args flags.Args) {
    57  	var longTitle string
    58  	if args.SiteTitle != "" && args.SiteDescription != "" {
    59  		longTitle = fmt.Sprintf("%s | %s", args.SiteTitle, args.SiteDescription)
    60  	}
    61  	values := map[string]interface{}{
    62  		"BaseURL":       args.BaseURL,
    63  		"SiteTitle":     args.SiteTitle,
    64  		"SiteTitleLong": longTitle,
    65  	}
    66  	funcs["gopages"] = func(defaultValue, firstKey string, keys ...string) (string, error) {
    67  		keys = append([]string{firstKey}, keys...) // require at least one key
    68  		for _, key := range keys {
    69  			value, ok := values[key]
    70  			var valueStr string
    71  			err := pipe.ChainFuncs(
    72  				func() error {
    73  					return pipe.ErrIf(!ok, errors.Errorf("Unknown gopages key: %q", key))
    74  				},
    75  				func() error {
    76  					var isString bool
    77  					valueStr, isString = value.(string)
    78  					return pipe.ErrIf(!isString, errors.Errorf("gopages key %q is not a string", key))
    79  				},
    80  			).Do()
    81  			if err != nil || valueStr != "" {
    82  				return template.HTMLEscapeString(valueStr), err
    83  			}
    84  		}
    85  		return defaultValue, nil
    86  	}
    87  }

    95  			return errors.Wrap(util.RemoveAll(fs, args.OutputPath), "Failed to clean output directory")
    96  		},
    97  		func() error {
    98  			return errors.Wrap(fs.MkdirAll(args.OutputPath, 0700), "Failed to create output directory")
    99  		},
   100  		func() error {

   106  		},
   107  		func() error {
   108  			modFS := &filesystemOpener{Filesystem: srcRoot}
   109  			ns.Bind(path.Join("/src", modulePackage), modFS, "/", vfs.BindReplace)
   110  
   111  			corpus = godoc.NewCorpus(ns)
   112  			return errors.Wrap(corpus.Init(), "Are there any Go files present? Failed to initialize corpus")
   113  		},
   114  	).Do()
   115  	if err != nil {
   116  		return err
   117  	}
   118  
   119  	pres := godoc.NewPresentation(corpus)
   120  	// attempt to override URLs for source code links
   121  	// TODO fix links from source pages back to docs
   122  	pres.URLForSrc = func(src string) string {
   123  		// seems godoc lib documentation is incorrect here, 'src' is actually the whole package path to the file
   124  		return path.Join(args.BaseURL, "/src", src)
   125  	}
   126  	pres.URLForSrcPos = func(src string, line, low, high int) string {
   127  		return (&url.URL{
   128  			Path:     path.Join(args.BaseURL, src),
   129  			Fragment: fmt.Sprintf("L%d", line),

   132  	pres.URLForSrcQuery = func(src, query string, line int) string {
   133  		return (&url.URL{
   134  			Path:     path.Join(args.BaseURL, src),
   135  			RawQuery: query,
   136  			Fragment: fmt.Sprintf("L%d", line),
   137  		}).String()
   138  	}
   139  	funcs := pres.FuncMap()
   140  	addGoPagesFuncs(funcs, args)
   141  	readTemplates(pres, funcs, ns)
   142  
   143  	// Generate all static assets and save to /lib/godoc
   144  	var ops []pipe.OpFunc
   145  	for name := range static.Files {
   146  		content := static.Files[name]
   147  		path := filepath.Join(args.OutputPath, "lib", "godoc", name)

   297  func redirect(url string) string {
   298  	var buf bytes.Buffer
   299  	err := template.Must(template.New("").Parse(`<!DOCTYPE html>
   300  <html>
   301  <head>
   302  <script>
   303  window.location = {{.URL}}
   304  </script>
   305  </head>
   306  <body>
   307  	<a href={{.URL}}>Click here to see this module's documentation.</a>
   308  </body>
   309  </html>
   310  `)).Execute(&buf, map[string]interface{}{
   311  		"URL": fmt.Sprintf("%q", url),
   312  	})
   313  	if err != nil {
   314  		panic(err)
   315  	}
   316  	return buf.String()

   335  func walkFiles(fs billy.Filesystem, path string, visit func(path string) error) error {
   336  	info, err := fs.Stat(path)
   337  	if err != nil {
   338  		return errors.Wrap(err, "Error looking up file")
   339  	}
   340  
   341  	if !info.IsDir() {
   342  		return visit(path)
   343  	}
   344  
   345  	dir, err := fs.ReadDir(path)
   346  	if err != nil {
   347  		return errors.Wrapf(err, "Error reading directory %q", path)
   348  	}
   349  	for _, info = range dir {
   350  		err := walkFiles(fs, filepath.Join(path, info.Name()), visit)
   351  		if err != nil {
   352  			return err
   353  		}
   354  	}
   355  	return nil
   356  }
