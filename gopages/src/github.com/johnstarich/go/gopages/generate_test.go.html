    18  func TestGenerateDocs(t *testing.T) {
    19  	// create a new package "thing" and generate docs for it
    20  	thing, err := ioutil.TempDir("", "")
    21  	require.NoError(t, err)
    22  	defer os.RemoveAll(thing)
    23  	wd, err := os.Getwd()
    24  	require.NoError(t, err)
    25  	defer func() {
    26  		require.NoError(t, os.Chdir(wd))
    27  	}()
    28  	require.NoError(t, os.Chdir(thing))
    29  
    30  	thingFS := osfs.New("")
    31  	outputFS := memfs.New()
    32  	writeFile := func(path, contents string) {
    33  		path = filepath.Join(thing, path)
    34  		err := os.MkdirAll(filepath.Dir(path), 0700)
    35  		require.NoError(t, err)
    36  		err = ioutil.WriteFile(path, []byte(contents), 0600)
    37  		require.NoError(t, err)
    38  	}
    39  
    40  	writeFile("go.mod", `module thing`)
    41  	writeFile("main.go", `
    42  package main
    43  
    44  func main() {
    45  	println("Hello world")
    46  }
    47  `)
    48  	writeFile("internal/hello/hello.go", `
    49  package lib
    50  
    51  // Hello says hello
    52  func Hello() {
    53  	println("Hello world")
    54  }
    55  `)
    56  
    57  	args := flags.Args{}
    58  	err = generateDocs(thing, "thing", args, thingFS, outputFS)
    59  	assert.NoError(t, err)
    60  
    61  	expectedDocs := []string{
    62  		"404.html",
    63  		"index.html",
    64  		"pkg/thing/index.html",
    65  		"pkg/thing/internal/hello/index.html",
    66  		"src/thing/internal/hello/hello.go.html",
    67  		"src/thing/main.go.html",
    68  	}
    69  	sort.Strings(expectedDocs)
    70  	var foundDocs []string
    71  	require.NoError(t, walkFiles(outputFS, "", func(path string) error {
    72  		if !strings.HasPrefix(path, filepath.Join("lib", "godoc")) {
    73  			foundDocs = append(foundDocs, path)
    74  		}
    75  		return nil
    76  	}))
    77  	sort.Strings(foundDocs)
    78  	assert.Equal(t, expectedDocs, foundDocs)
    79  
    80  	f, err := outputFS.Open("pkg/thing/index.html")
    81  	require.NoError(t, err)
    82  	indexContents, err := ioutil.ReadAll(f)
    83  	require.NoError(t, err)
    84  	assert.Contains(t, string(indexContents), "internal/hello")
    85  }
