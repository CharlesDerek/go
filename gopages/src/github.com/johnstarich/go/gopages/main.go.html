    66  func run(modulePath string, args flags.Args) error {
    67  	goMod := filepath.Join(modulePath, "go.mod")
    68  	var modulePackage string
    69  	err := pipe.ChainFuncs(
    70  		func() error {
    71  			_, err := os.Stat(goMod)
    72  			return pipe.ErrIf(os.IsNotExist(err), errors.New("go.mod not found in the current directory"))
    73  		},
    74  		func() error {
    75  			buf, err := ioutil.ReadFile(goMod)
    76  			modulePackage = modfile.ModulePath(buf)
    77  			return err
    78  		},
    79  		func() error {
    80  			return pipe.ErrIf(modulePackage == "", errors.Errorf("Unable to find module package name in go.mod file: %s", goMod))
    81  		},
    82  	).Do()
    83  	if err != nil {
    84  		return err
    85  	}
    86  
    87  	fmt.Println("Generating godoc static pages for module...", modulePackage)
    88  
    89  	if !args.GitHubPages {
    90  		fs := osfs.New("")
    91  		return generate.Docs(modulePath, modulePackage, fs, fs, args)
    92  	}
    93  
    94  	var repoRoot, remote string
    95  	var absOutputPath string
    96  	err = pipe.ChainFuncs(
    97  		func() error {
    98  			var err error
    99  			repoRoot, remote, err = getCurrentPathAndRemote(modulePath)
   100  			return err
   101  		},
   102  		func() error {
   103  			var err error
   104  			absOutputPath, err = filepath.Abs(args.OutputPath)
   105  			return errors.Wrap(err, "Failed to get absolute path of output directory")
   106  		},
   107  		func() error {
   108  			var err error
   109  			args.OutputPath, err = filepath.Rel(repoRoot, absOutputPath)
   110  			return errors.Wrap(err, "Output path must be inside repository for gh-pages integration")
   111  		},
   112  	).Do()
   113  	if err != nil {
   114  		return err
   115  	}
   116  
   117  	src := osfs.New("")
   118  	fs := memfs.New()
   119  
   120  	var repo *git.Repository
   121  	var workTree *git.Worktree
   122  	err = pipe.ChainFuncs(
   123  		func() error {
   124  			var err error
   125  			repo, err = git.Clone(memory.NewStorage(), fs, &git.CloneOptions{
   126  				URL:           remote,
   127  				ReferenceName: plumbing.NewBranchReferenceName(ghPagesBranch),
   128  				SingleBranch:  true,
   129  			})
   130  			return errors.Wrap(err, "Failed to clone in-memory copy of repo. Be sure the 'gh-pages' orphaned branch exists: https://help.github.com/en/github/working-with-github-pages/creating-a-github-pages-site-with-jekyll#creating-your-site")
   131  		},
   132  		func() error {
   133  			var err error
   134  			workTree, err = repo.Worktree()
   135  			return err
   136  		},
   137  		func() error {
   138  			_, _ = workTree.Remove(args.OutputPath) // remove old files on a best-effort basis. if the path doesn't exist, it could error
   139  			return nil
   140  		},
   141  		func() error {
   142  			return generate.Docs(modulePath, modulePackage, src, fs, args)
   143  		},
   144  	).Do()
   145  	if err != nil {
   146  		return err
   147  	}
   148  
   149  	fmt.Println("Committing and pushing changes to gh-pages branch...")
   150  
   151  	return pipe.ChainFuncs(
   152  		func() error {
   153  			_, err := workTree.Add(args.OutputPath)
   154  			return errors.Wrap(err, "Failed to add output dir to git")
   155  		},
   156  		func() error {
   157  			commitMessage := "Update GoPages"
   158  			if args.SiteTitle != "" {
   159  				commitMessage += ": " + args.SiteTitle
   160  			}
   161  			_, err := workTree.Commit(commitMessage, &git.CommitOptions{
   162  				Author: commitAuthor(),
   163  			})
   164  			return errors.Wrap(err, "Failed to commit gopages files")
   165  		},
   166  		func() error {
   167  			pushOpts := &git.PushOptions{}
   168  			if args.GitHubPagesUser != "" || args.GitHubPagesToken != "" {
   169  				pushOpts.Auth = &gitHTTP.BasicAuth{Username: args.GitHubPagesUser, Password: args.GitHubPagesToken}
   170  			}
   171  			err = repo.Push(pushOpts)
   172  			return errors.Wrap(err, "Failed to push gopages commit")
   173  		},
   174  	).Do()

   177  func getCurrentPathAndRemote(repoPath string) (string, string, error) {
   178  	repo, err := git.PlainOpenWithOptions(repoPath, &git.PlainOpenOptions{
   179  		DetectDotGit: true,
   180  	})
   181  	if err != nil {
   182  		return "", "", errors.Wrapf(err, "Failed to open repo at %q", repoPath)
   183  	}
   184  
   185  	fs, err := repo.Worktree()
   186  	if err != nil {
   187  		return "", "", errors.Wrap(err, "Failed to set up work tree for repo")
   188  	}
   189  	repoRoot, err := filepath.EvalSymlinks(fs.Filesystem.Root())
   190  	if err != nil {
   191  		return "", "", err
   192  	}
   193  
   194  	remote, err := repo.Remote(git.DefaultRemoteName)
   195  	remoteURL := remote.Config().URLs[0]
   196  	return repoRoot, remoteURL, errors.Wrap(err, "Failed to get repo remote")
   197  }
