    67  func run(modulePath string, args flags.Args) error {
    68  	var modulePackage string
    69  	var linker source.Linker
    70  	err := pipe.ChainFuncs(
    71  		func() error {
    72  			var err error
    73  			modulePackage, err = module.Package(modulePath)
    74  			return err
    75  		},
    76  		func() error {
    77  			var err error
    78  			linker, err = args.Linker(modulePackage)
    79  			return err
    80  		},
    81  	).Do()
    82  	if err != nil {
    83  		return err
    84  	}
    85  
    86  	fmt.Println("Generating godoc static pages for module...", modulePackage)
    87  	if !args.GitHubPages {
    88  		fs := osfs.New("")
    89  		return generate.Docs(modulePath, modulePackage, fs, fs, args, linker)
    90  	}
    91  
    92  	var repoRoot, remote string
    93  	var absOutputPath string
    94  	err = pipe.ChainFuncs(
    95  		func() error {
    96  			var err error
    97  			repoRoot, remote, err = getCurrentPathAndRemote(modulePath)
    98  			return err
    99  		},
   100  		func() error {
   101  			var err error
   102  			absOutputPath, err = filepath.Abs(args.OutputPath)
   103  			return errors.Wrap(err, "Failed to get absolute path of output directory")
   104  		},
   105  		func() error {
   106  			var err error
   107  			args.OutputPath, err = filepath.Rel(repoRoot, absOutputPath)
   108  			return errors.Wrap(err, "Output path must be inside repository for gh-pages integration")
   109  		},
   110  	).Do()
   111  	if err != nil {
   112  		return err
   113  	}
   114  
   115  	src := osfs.New("")
   116  	fs := memfs.New()
   117  
   118  	var repo *git.Repository
   119  	var workTree *git.Worktree
   120  	err = pipe.ChainFuncs(
   121  		func() error {
   122  			var err error
   123  			repo, err = git.Clone(memory.NewStorage(), fs, &git.CloneOptions{
   124  				URL:           remote,
   125  				ReferenceName: plumbing.NewBranchReferenceName(ghPagesBranch),
   126  				SingleBranch:  true,
   127  			})
   128  			return errors.Wrap(err, "Failed to clone in-memory copy of repo. Be sure the 'gh-pages' orphaned branch exists: https://help.github.com/en/github/working-with-github-pages/creating-a-github-pages-site-with-jekyll#creating-your-site")
   129  		},
   130  		func() error {
   131  			var err error
   132  			workTree, err = repo.Worktree()
   133  			return err
   134  		},
   135  		func() error {
   136  			_, _ = workTree.Remove(args.OutputPath) // remove old files on a best-effort basis. if the path doesn't exist, it could error
   137  			return nil
   138  		},
   139  		func() error {
   140  			return generate.Docs(modulePath, modulePackage, src, fs, args, linker)
   141  		},
   142  	).Do()
   143  	if err != nil {
   144  		return err
   145  	}
   146  
   147  	fmt.Println("Committing and pushing changes to gh-pages branch...")
   148  
   149  	return pipe.ChainFuncs(
   150  		func() error {
   151  			_, err := workTree.Add(args.OutputPath)
   152  			return errors.Wrap(err, "Failed to add output dir to git")
   153  		},
   154  		func() error {
   155  			commitMessage := "Update GoPages"
   156  			if args.SiteTitle != "" {
   157  				commitMessage += ": " + args.SiteTitle
   158  			}
   159  			_, err := workTree.Commit(commitMessage, &git.CommitOptions{
   160  				Author: commitAuthor(),
   161  			})
   162  			return errors.Wrap(err, "Failed to commit gopages files")
   163  		},
   164  		func() error {
   165  			pushOpts := &git.PushOptions{}
   166  			if args.GitHubPagesUser != "" || args.GitHubPagesToken != "" {
   167  				pushOpts.Auth = &gitHTTP.BasicAuth{Username: args.GitHubPagesUser, Password: args.GitHubPagesToken}
   168  			}
   169  			err = repo.Push(pushOpts)
   170  			return errors.Wrap(err, "Failed to push gopages commit")
   171  		},
   172  	).Do()
   173  }

   175  func getCurrentPathAndRemote(repoPath string) (string, string, error) {
   176  	var repo *git.Repository
   177  	var fs *git.Worktree
   178  	var repoRoot string
   179  	var remoteURL string
   180  	err := pipe.ChainFuncs(
   181  		func() error {
   182  			var err error
   183  			repo, err = git.PlainOpenWithOptions(repoPath, &git.PlainOpenOptions{
   184  				DetectDotGit: true,
   185  			})
   186  			return errors.Wrapf(err, "Failed to open repo at %q", repoPath)
   187  		},
   188  		func() error {
   189  			var err error
   190  			fs, err = repo.Worktree()
   191  			return errors.Wrap(err, "Failed to set up work tree for repo")
   192  		},
   193  		func() error {
   194  			var err error
   195  			repoRoot, err = filepath.EvalSymlinks(fs.Filesystem.Root())
   196  			return err
   197  		},
   198  		func() error {
   199  			remote, err := repo.Remote(git.DefaultRemoteName)
   200  			remoteURL = remote.Config().URLs[0]
   201  			return errors.Wrap(err, "Failed to get repo remote")
   202  		},
   203  	).Do()
   204  	return repoRoot, remoteURL, err
   205  }
