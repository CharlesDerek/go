    65  func run(modulePath string, args flags.Args) error {
    66  	goMod := filepath.Join(modulePath, "go.mod")
    67  	var modulePackage string
    68  	err := pipe.ChainFuncs(
    69  		func() error {
    70  			_, err := os.Stat(goMod)
    71  			return pipe.ErrIf(os.IsNotExist(err), errors.New("go.mod not found in the current directory"))
    72  		},
    73  		func() error {
    74  			buf, err := ioutil.ReadFile(goMod)
    75  			modulePackage = modfile.ModulePath(buf)
    76  			return err
    77  		},
    78  		func() error {
    79  			return pipe.ErrIf(modulePackage == "", errors.Errorf("Unable to find module package name in go.mod file: %s", goMod))
    80  		},
    81  	).Do()
    82  	if err != nil {
    83  		return err
    84  	}
    85  
    86  	fmt.Println("Generating godoc static pages for module...", modulePackage)
    87  
    88  	if !args.GitHubPages {
    89  		fs := osfs.New("")
    90  		return generateDocs(modulePath, modulePackage, args, fs, fs)
    91  	}
    92  
    93  	var repoRoot, remote string
    94  	var absOutputPath string
    95  	err = pipe.ChainFuncs(
    96  		func() error {
    97  			var err error
    98  			repoRoot, remote, err = getCurrentPathAndRemote(modulePath)
    99  			return err
   100  		},
   101  		func() error {
   102  			var err error
   103  			absOutputPath, err = filepath.Abs(args.OutputPath)
   104  			return errors.Wrap(err, "Failed to get absolute path of output directory")
   105  		},
   106  		func() error {
   107  			var err error
   108  			args.OutputPath, err = filepath.Rel(repoRoot, absOutputPath)
   109  			return errors.Wrap(err, "Output path must be inside repository for gh-pages integration")
   110  		},
   111  	).Do()
   112  	if err != nil {
   113  		return err
   114  	}
   115  
   116  	src := osfs.New("")
   117  	fs := memfs.New()
   118  
   119  	var repo *git.Repository
   120  	var workTree *git.Worktree
   121  	err = pipe.ChainFuncs(
   122  		func() error {
   123  			var err error
   124  			repo, err = git.Clone(memory.NewStorage(), fs, &git.CloneOptions{
   125  				URL:           remote,
   126  				ReferenceName: plumbing.NewBranchReferenceName(ghPagesBranch),
   127  				SingleBranch:  true,
   128  			})
   129  			return errors.Wrap(err, "Failed to clone in-memory copy of repo. Be sure the 'gh-pages' orphaned branch exists: https://help.github.com/en/github/working-with-github-pages/creating-a-github-pages-site-with-jekyll#creating-your-site")
   130  		},
   131  		func() error {
   132  			var err error
   133  			workTree, err = repo.Worktree()
   134  			return err
   135  		},
   136  		func() error {
   137  			_, _ = workTree.Remove(args.OutputPath) // remove old files on a best-effort basis. if the path doesn't exist, it could error
   138  			return nil
   139  		},
   140  		func() error {
   141  			return generateDocs(modulePath, modulePackage, args, src, fs)
   142  		},
   143  	).Do()
   144  	if err != nil {
   145  		return err
   146  	}
   147  
   148  	fmt.Println("Committing and pushing changes to gh-pages branch...")
   149  
   150  	return pipe.ChainFuncs(
   151  		func() error {
   152  			_, err := workTree.Add(args.OutputPath)
   153  			return errors.Wrap(err, "Failed to add output dir to git")
   154  		},
   155  		func() error {
   156  			commitMessage := "Update GoPages"
   157  			if args.SiteTitle != "" {
   158  				commitMessage += ": " + args.SiteTitle
   159  			}
   160  			_, err := workTree.Commit(commitMessage, &git.CommitOptions{
   161  				Author: commitAuthor(),
   162  			})
   163  			return errors.Wrap(err, "Failed to commit gopages files")
   164  		},
   165  		func() error {
   166  			pushOpts := &git.PushOptions{}
   167  			if args.GitHubPagesUser != "" || args.GitHubPagesToken != "" {
   168  				pushOpts.Auth = &gitHTTP.BasicAuth{Username: args.GitHubPagesUser, Password: args.GitHubPagesToken}
   169  			}
   170  			err = repo.Push(pushOpts)
   171  			return errors.Wrap(err, "Failed to push gopages commit")
   172  		},
   173  	).Do()
   174  }

   176  func getCurrentPathAndRemote(repoPath string) (string, string, error) {
   177  	repo, err := git.PlainOpenWithOptions(repoPath, &git.PlainOpenOptions{
   178  		DetectDotGit: true,
   179  	})
   180  	if err != nil {
   181  		return "", "", errors.Wrapf(err, "Failed to open repo at %q", repoPath)
   182  	}
   183  
   184  	fs, err := repo.Worktree()
   185  	if err != nil {
   186  		return "", "", errors.Wrap(err, "Failed to set up work tree for repo")
   187  	}
   188  	repoRoot, err := filepath.EvalSymlinks(fs.Filesystem.Root())
   189  	if err != nil {
   190  		return "", "", err
   191  	}
   192  
   193  	remote, err := repo.Remote(git.DefaultRemoteName)
   194  	remoteURL := remote.Config().URLs[0]
   195  	return repoRoot, remoteURL, errors.Wrap(err, "Failed to get repo remote")
   196  }
