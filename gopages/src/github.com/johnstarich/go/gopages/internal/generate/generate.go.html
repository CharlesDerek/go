   346  func walkFilesFn(fs billy.Filesystem, path string, visit func(path string, isDir bool) error) error {
   347  	info, err := fs.Lstat(path)
   348  	if err != nil {
   349  		return errors.Wrap(err, "Error looking up file")
   350  	}
   351  
   352  	if info.Mode()&os.ModeSymlink == os.ModeSymlink {
   353  		// ignore symlinks to avoid infinite recursion
   354  		return nil
   355  	}
   356  
   357  	isDir := info.IsDir()
   358  	if err := visit(path, isDir); err != nil {
   359  		return err
   360  	}
   361  
   362  	if !isDir {
   363  		return nil
   364  	}
   365  
   366  	dir, err := fs.ReadDir(path)
   367  	if err != nil {
   368  		return errors.Wrapf(err, "Error reading directory %q", path)
   369  	}
   370  	for _, info = range dir {
   371  		err := walkFilesFn(fs, filepath.Join(path, info.Name()), visit)
   372  		if err == filepath.SkipDir {
   373  			if !info.IsDir() {
   374  				break // for SkipDir on a file, skip remaining files in directory
   375  			}
   376  			// otherwise continue recursing other files in this dir
   377  		} else if err != nil {
   378  			return err
   379  		}
   380  	}
   381  	return nil
   382  }
