    15  func addGoPagesFuncs(funcs template.FuncMap, modulePackage string, args flags.Args) {
    16  	var longTitle string
    17  	if args.SiteTitle != "" && args.SiteDescription != "" {
    18  		longTitle = fmt.Sprintf("%s | %s", args.SiteTitle, args.SiteDescription)
    19  	}
    20  	values := map[string]interface{}{
    21  		"BaseURL":       args.BaseURL,
    22  		"ModuleURL":     path.Join(args.BaseURL, "pkg", modulePackage) + "/",
    23  		"SiteTitle":     args.SiteTitle,
    24  		"SiteTitleLong": longTitle,
    25  	}
    26  	funcs["gopages"] = func(defaultValue, firstKey string, keys ...string) (string, error) {
    27  		keys = append([]string{firstKey}, keys...) // require at least one key
    28  		for _, key := range keys {
    29  			value, ok := values[key]
    30  			var valueStr string
    31  			err := pipe.ChainFuncs(
    32  				func() error {
    33  					return pipe.ErrIf(!ok, errors.Errorf("Unknown gopages key: %q", key))
    34  				},
    35  				func() error {
    36  					var isString bool
    37  					valueStr, isString = value.(string)
    38  					return pipe.ErrIf(!isString, errors.Errorf("gopages key %q is not a string", key))
    39  				},
    40  			).Do()
    41  			if err != nil || valueStr != "" {
    42  				return template.HTMLEscapeString(valueStr), err
    43  			}
    44  		}
    45  		return defaultValue, nil
    46  	}
