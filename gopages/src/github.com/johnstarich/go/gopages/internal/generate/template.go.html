    16  func addGoPagesFuncs(funcs template.FuncMap, modulePackage string, args flags.Args) {
    17  	var longTitle string
    18  	if args.SiteTitle != "" && args.SiteDescription != "" {
    19  		longTitle = fmt.Sprintf("%s | %s", args.SiteTitle, args.SiteDescription)
    20  	}
    21  	values := map[string]interface{}{
    22  		"BaseURL":       args.BaseURL,
    23  		"ModuleURL":     path.Join(args.BaseURL, "/pkg", modulePackage) + "/",
    24  		"SiteTitle":     args.SiteTitle,
    25  		"SiteTitleLong": longTitle,
    26  	}
    27  	funcs["gopages"] = func(defaultValue, firstKey string, keys ...string) (string, error) {
    28  		keys = append([]string{firstKey}, keys...) // require at least one key
    29  		for _, key := range keys {
    30  			value, ok := values[key]
    31  			var valueStr string
    32  			err := pipe.ChainFuncs(
    33  				func() error {
    34  					return pipe.ErrIf(!ok, errors.Errorf("Unknown gopages key: %q", key))
    35  				},
    36  				func() error {
    37  					var isString bool
    38  					valueStr, isString = value.(string)
    39  					return pipe.ErrIf(!isString, errors.Errorf("gopages key %q is not a string", key))
    40  				},
    41  			).Do()
    42  			if err != nil || valueStr != "" {
    43  				return template.HTMLEscapeString(valueStr), err
    44  			}
    45  		}
    46  		return defaultValue, nil
    47  	}
    48  	funcs["gopagesWatchScript"] = func() string {
    49  		if !args.Watch {
    50  			return ""
    51  		}
    52  		return fmt.Sprintf(`
    53  <script>
    54  const startDate = %q
    55  const timeoutMillis = 2000
    56  const poll = () => {
    57  	fetch(window.location).then(resp => {
    58  		const newDate = resp.headers.get("GoPages-Last-Updated")
    59  		if (newDate != startDate) {
    60  			window.location.reload()
    61  		}
    62  	})
    63  }
    64  window.setInterval(poll, timeoutMillis)
    65  </script>
    66  `, time.Now().Format(time.RFC3339))
    67  	}
    68  }
